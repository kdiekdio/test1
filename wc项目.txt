void signalcount(char *filename)
{
	int len = 0, i = 0, signline = 0;
	FILE *fp;
	char str[1000];
	fp = fopen(filename, "r");
	if (fp == NULL)
	{
		printf("请在文件中添加内容\n");
		exit(-1);
	}
	while (!feof(fp))
	{
		fgets(str, sizeof(str), fp);
		len = strlen(str);
		for (i = 0; i<len; i++)
		{
			if ((str[i] == '/'&&str[i + 1] == '/') || (str[i] == '/'&&str[i + 1] == '*'))
			{
				signline++;
				break;
			}
		}
	}
	fclose(fp);
	printf("注释行数%d行\n", signline);
}
